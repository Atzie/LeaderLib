PROC
LLLIB_Updater_ClearOutdatedDatabases()
AND
DB_Registered_Mod((STRING)_ModName, (STRING)_Version, "LaughingLeader")
THEN
NOT DB_Registered_Mod(_ModName, _Version, "LaughingLeader");
//END_REGION

//REGION UPDATE_EVENT
IF
GameStarted(_,_)
THEN
LeaderLog_Log("SYSTEM", "[LLLIB:Updater] Game started.");
LLLIB_Updater_ClearOutdatedDatabases();

IF
GameStarted(_,_)
AND
LeaderUpdater_QRY_ModUpdateNeeded("LeaderLib", "LaughingLeader", "0.9.0.0")
THEN
LeaderLog_Log("SYSTEM", "[LLLIB:Updater] Updating LeaderLib to version [0.9.0.0].");
LeaderUpdater_Register_Mod("LeaderLib", "LaughingLeader", "0.9.0.0");
LeaderLib_SendEvent("LLLIB_Events_OnUpdateSettings");
//END_REGION

//REGION UNINSTALLING
PROC
LeaderUpdater_UnregisterMod("LeaderLib", "LaughingLeader")
THEN
GlobalClearFlag("LeaderLib_Active");

PROC
LeaderUpdater_UninstallMod("LeaderLib", "LaughingLeader")
THEN
LeaderLog_Log("SYSTEM", "[LLLIB:Updater:Uninstaller] Uninstalling LeaderLib.");
LeaderLog_Log("SYSTEM", "[LLLIB:Updater:Uninstaller] Attempting to end all LeaderLib dialogs. Resuming in 2 seconds.");
LLLIB_Updater_CloseDialogs();
TimerLaunch("LLLIB_Timers_Updater_UninstallerPhase2", 2000);

IF
TimerFinished("LLLIB_Timers_Updater_UninstallerPhase2")
THEN
LeaderLog_Log("SYSTEM", "[LLLIB:Updater:Uninstaller] Clearing all flags.");
LLLIB_Updater_ClearFlags();
LeaderLog_Log("SYSTEM", "[LLLIB:Updater:Uninstaller] Destroying all LeaderLib items.");
LLLIB_Updater_DestroyItems();
LeaderLog_Log("SYSTEM", "[LLLIB:Updater:Uninstaller] Clearing databases.");
LeaderUpdater_SysClear_AllDatabases("LeaderLib", "LaughingLeader");
LeaderLog_Log("SYSTEM", "[LLLIB:Updater:Uninstaller] Unregistering LeaderLib.");
LeaderUpdater_UnregisterMod("LeaderLib", "LaughingLeader");
