//REGION UPDATING
PROC
LLLIB_System_UpdateDatabases()
THEN
LeaderLog_Log("DEBUG", "[LLLIB:Updater] Databases updated.");

PROC
LLLIB_System_RegisterDatabases()
THEN
LeaderLog_Log("DEBUG", "[LLLIB:Updater] Databases registered.");

PROC
LLLIB_Updater_ResetDatabases()
THEN
LeaderUpdater_ClearAllDatabases("LeaderLib", "LaughingLeader");
LLLIB_System_UpdateDatabases();
LeaderLog_Log("DEBUG", "[LLLIB:Updater] Databases cleared and re-initialized.");
//END_REGION

//REGION UPDATE_EVENT
IF
GameStarted(_,_)
THEN
LeaderLog_Log("DEBUG", "[LLLIB:Updater] Game started.");

IF
GameStarted(_,_)
AND
LeaderUpdater_QRY_ModUpdateNeeded("LeaderLib", "0.7.0.0", "LaughingLeader")
THEN
LeaderLog_Log("DEBUG", "[LLLIB:Updater] Updating LeaderLib.");
LeaderUpdater_Register_Mod("LeaderLib", "0.7.0.0", "LaughingLeader");
LLLIB_System_UpdateDatabases();
LeaderLib_SendEvent("LeaderLib_Events_OnDatabasesUpdated");
//END_REGION

//REGION UNINSTALLING
PROC
LeaderUpdater_UnregisterMod("LeaderLib", "LaughingLeader")
THEN
GlobalClearFlag("LeaderLib_Active");

PROC
LeaderUpdater_UninstallMod("LeaderLib", "LaughingLeader")
THEN
LeaderLog_Log("DEBUG", "[LLLIB:Updater:Uninstaller] Uninstalling LeaderLib.");
LeaderLog_Log("DEBUG", "[LLLIB:Updater:Uninstaller] Attempting to end all LeaderLib dialogs. Resuming in 2 seconds.");
LLLIB_Updater_CloseDialogs();
TimerLaunch("LLLIB_Timers_Updater_UninstallerPhase2", 2000);

IF
TimerFinished("LLLIB_Timers_Updater_UninstallerPhase2")
THEN
LeaderLog_Log("DEBUG", "[LLLIB:Updater:Uninstaller] Clearing all flags.");
LLLIB_Updater_ClearFlags();
LeaderLog_Log("DEBUG", "[LLLIB:Updater:Uninstaller] Destroying all LeaderLib items.");
LLLIB_Updater_DestroyItems();
LeaderLog_Log("DEBUG", "[LLLIB:Updater:Uninstaller] Clearing databases.");
LeaderUpdater_ClearAllDatabases("LeaderLib", "LaughingLeader");
LeaderLog_Log("DEBUG", "[LLLIB:Updater:Uninstaller] Unregistering LeaderLib.");
//LeaderUpdater_UnregisterMod();
